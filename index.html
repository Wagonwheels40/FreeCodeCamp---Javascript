// Store a name and age using the assignment operator var name = "Gareth"; var age = 26; // Display the stored values console.log("My name is " + name + " and I am " + age + " years old."); // Change the stored values name = "George" age = 25; // Display the updated values console.log("My name is " + name + " and I am " + age + " years old."); /* Storing Values with the Assignment Operator */
